<script>
const els = document.querySelectorAll('.fade');
window.addEventListener('scroll',()=>{
els.forEach(el=>{
if(el.getBoundingClientRect().top < window.innerHeight-50){
el.classList.add('show');
}
});

function toggleTheme(){
document.body.classList.toggle('light');
document.body.classList.toggle('dark');
}

const questions = [
{q:"Машиналық оқыту не?",a:["Деректер арқылы үйрену","Ойын"],c:0},
{q:"Бұл не бөлігі?",a:["ЖИ","Графика"],c:0},
{q:"Қайсысы түрі?",a:["Бақыланатын","Paint"],c:0}
];

let i=0, score=0;

function showQ(){
document.getElementById("q").innerText = questions[i].q;
let ans="";
questions[i].a.forEach((x,index)=>{
ans+=`<button onclick="check(${index})">${x}</button>`;
});
document.getElementById("answers").innerHTML = ans;
document.getElementById("score").innerText = "Ұпай: "+score;
}

function check(index){
if(index===questions[i].c){ score++; }
}

function nextQ(){
i++;
if(i<questions.length){ showQ(); }
else{
document.getElementById("q").innerText="Тест бітті!";
document.getElementById("answers").innerHTML="";
document.getElementById("score").innerText="Қорытынды ұпай: "+score;
}
}

showQ();
</script>
